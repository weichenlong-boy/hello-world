XGBoost特征构造

本文利用XGBoost构造特征，返回是每棵树的叶子节点序列，构造的特征可以用于Logistic Regression和LibFFM，下面代码实现：

from sklearn.model_selection import cross_val_score
from sklearn.preprocessing import OneHotEncoder
from xgboost.sklearn import XGBClassifier

class XgboostFeature():
    '''
    可以传入xgboost的参数
    常用传入特征的个数 即树的个数 默认30
    name表示构造之后的特征用于LR还是LibFFM
    '''
    def __init__(self,name='LR',n_estimators=30,learning_rate =0.3,max_depth=3,min_child_weight=1,gamma=0.3,subsample=0.8,colsample_bytree=0.8,objective= 'binary:logistic',nthread=4,scale_pos_weight=1,reg_alpha=1e-05,reg_lambda=1,seed=27):
        self.n_estimators=n_estimators
        self.learning_rate=learning_rate
        self.max_depth=max_depth
        self.min_child_weight=min_child_weight
        self.gamma=gamma
        self.subsample=subsample
        self.colsample_bytree=colsample_bytree
        self.objective=objective
        self.nthread=nthread
        self.scale_pos_weight=scale_pos_weight
        self.reg_alpha=reg_alpha
        self.reg_lambda=reg_lambda
        self.seed=seed
        self.name=name
        print ('Xgboost Feature start, new_feature number:',n_estimators)
    def LR(self,X,Y):
        onehot = OneHotEncoder()
        onehot_X = onehot.fit_transform(X)
        onehot_Y = onehot.transform(Y)
        return onehot_X,onehot_Y
    ##整体训练
    def fit_model(self,X_train,y_train,X_test):
        clf = XGBClassifier(
             learning_rate =self.learning_rate,
             n_estimators=self.n_estimators,
             max_depth=self.max_depth,
             min_child_weight=self.min_child_weight,
             gamma=self.gamma,
             subsample=self.subsample,
             colsample_bytree=self.colsample_bytree,
             objective= self.objective,
             nthread=self.nthread,
             scale_pos_weight=self.scale_pos_weight,
             reg_alpha=self.reg_alpha,
             reg_lambda=self.reg_lambda,
             seed=self.seed)
        clf.fit(X_train, y_train)
        auc = cross_val_score(clf, X_train, y_train, scoring='roc_auc', cv=10)
        print (f'XGBoost交叉验证分数:{auc.mean()}')
        new_feature= clf.apply(X_train)
        new_feature_test= clf.apply(X_test)
        if self.name == 'LR':
            onehot_x,onehot_y = self.LR(new_feature,new_feature_test)
            print ("Training set sample number remains the same")
            return onehot_x,onehot_y
        else:
            return new_feature,new_feature_test

调用示例，name为'LR'，即是onehot后的用于Logistic Regression

xgb = XgboostFeature(name='LR',n_estimators=100)
X_train_new,X_test_new = xgb.fit_model(train,label,test)
